\section{Simple Double Integral}

\subsection{Linear Function}

\begin{theorem}
\[
	\iint_D x \dd x \dd y = \overline{x} A(D)
	\quad
	\iint_D y \dd x \dd y = \overline{y} A(D)
\] 
where $A(D)$ is the area of the domain $D$ and $\overline{x}$ and $\overline{y}$ are
the average values of $x$ and $y$ over the domain $D$. Or so called the geometric 
center of the domain $D$.

\end{theorem}

\begin{example}
	Region $D$ is bounded by the curves $x=-\sqrt{2y-y^2}$ and $x=-2, y=0, y=2$, 
	try to calculate the integral
	 \[
		I = \iint_D y \dd x \dd y
	\] 
	\begin{answer}
\begin{mathfig}
	I = \overline{y} A(D) = 4 - \frac{\pi}{2}

\section{Symmetry}

\begin{lemma}
Suppose a domain $D$ is described by constraints $D(x, y)$ denoting the equations or inequalities
that define the domain, we can define a new domain $D'(x, y) = D(y, x)$ by exchanging $x$ and $y$.
Then we have
\[
	\iint_D f(x, y) \dd \sigma = \iint_{D'} f(y, x) \dd \sigma
\] 
\end{lemma}
\begin{theorem}[For all xy symmetric domain]
	\label{thm:xy-symmetric-domain-exchange}
When the domain $D$ is symmetric about the $y=x$, or in other words, exchanging
$x$ and $y$ does not change the domain, $D=D'$, then we have
\[
	\iint_D f(x, y) \dd \sigma = \iint_{D} f(y, x) \dd \sigma
	= \frac{1}{2} \iint_D \left[ f(x, y) + f(y, x) \right] \dd \sigma
\] 
\end{theorem}

\begin{example}
	\[
		I = \int_0^1 \dd x \int_{1-x}^{2-x} e^{(x+y)^2} (\sin^2 x + \cos^2 y) \dd y
		+\int_1^2 \dd x \int_{0}^{2-x} e^{(x+y)^2} (\sin^2 x + \cos^2 y) \dd y
	\] 
	\begin{answer}
		\begin{mathfig}
				I &= \iint_D e^{(x+y)^2} (\sin^2 x + \cos^2 y) \dd \sigma \\
				&= \frac{1}{2} \iint_D e^{(x+y)^2} (\sin^2 x + \cos^2 y + \sin^2 y + \cos^2 x) \dd \sigma \\
				&= \frac{1}{2} \iint_D e^{(x+y)^2} (1 + 1) \dd \sigma \\
				&= \iint_D e^{(x+y)^2} \dd \sigma
			\tikzfig
				% axis

\begin{theorem}

If the domain $D$ is symmetric about the line $y=x$, then we can divide the domain
into two parts $D_1$ and $D_2$ symmetrically,
(its arbitrary to choose which part is $D_1$ or $D_2$)
then we have

\begin{mathfig}

	&= \iint_{D_1} f(x, y) \dd \sigma + \iint_{D_2} f(x, y) \dd \sigma \\
	&= \iint_{D_1} f(x, y) \dd \sigma + \iint_{D_1} f(y, x) \dd \sigma \\
	&= \iint_{D_1} [f(x, y) + f(y, x)] \dd \sigma \\
	\tikzfig[scale=1.5]

which is more convenient to calculate the integral since we only need to integrate
over half of the domain, dont need to consider piecewise boundary conditions.
	
\end{theorem}

\begin{example}
	Let $D:\sqrt{1-y^2} \leq x \leq 1, -1 \leq y \leq 1$, try to calculate the integral
	\[

				&= \iint_{D_{11}} \frac{x+y}{\sqrt{x^2 + y^2}} \dd x \dd y \\
				&= \int_0^{\pi/4} \dd \theta \int_1^{1/\cos \theta} 
				(\cos \theta + \sin \theta) r\dd r \\
				&= \int_0^{\pi/4} \frac{1}{2} (\cos \theta + \sin \theta)\left(
					\frac{1}{\cos^2 \theta}-1\right) \dd \theta

			\fill[blue!20] (0,1) arc (90:-90:1) -- (1,-1) -- (1,1) -- cycle;
			% fill upper half domain as D_1
			\fill[cyan!20,opacity=.8] (0,1) arc (90:0:1) -- (1,0) -- (1,1) -- cycle;
			% fill lower half of D_1 as D_11
			\fill[orange!20,opacity=.8] (1,0) arc (0:45:1) -- (1,1) -- cycle;
			% circle use arc to avoid endpoint problem

			\draw[blue] (0,1) -- (1,1) -- (1,-1) -- (0,-1);

\begin{theorem}
	If the domain $D$ is symmetric about the line $y=x$, and the integrand function $f(x, y)$
	can be decomposed into independent two functions
	 \[
		f(x, y) = g(x) + h(y)
	\] 
	then we have
	\[
		\iint_D f \dd \sigma = 
		\iint_D g(x) + h(x) \dd \sigma = \iint_D g(y) + h(y) \dd \sigma
	\] 
\end{theorem}
\begin{proof}
	By \ref{thm:xy-symmetric-domain-exchange}, we have
	\[
		\iint_D h(y) \dd \sigma = \iint_D h(x) \dd \sigma
	\] 
\end{proof}

\begin{example}
	Let $D: \sqrt{|x|} + \sqrt{|y|} \leq 1$, try to calculate the integral
	\[
		I = \iint_D 2\sqrt{|x|} + 3\sqrt{|y|} \dd x \dd y
	\] 
	\begin{answer}
		Let $D_1$ be the part of $D$ in the first quadrant, then we have
		\[
			I = 4\iint_{D_1} 2\sqrt{x} + 3\sqrt{y} \dd x \dd y
		\] 
		$D_1$ is symmetric about the line $y=x$, so we have
		\[
			I = 4\iint_{D_1} 2\sqrt{x} + 3\sqrt{x} \dd x \dd y
			= 20 \iint_{D_1} \sqrt{x} \dd x \dd y
		\] 
		Let $x=u^2, y= v^2$ and new $D'$ is bounded by $u+v=1, u \geq 0, v \geq 0$, then we have
		 \[
			I = 20 \iint_{D'} u \cdot 2u \cdot 2v \dd u \dd v 
			= 40 \int_0^1 u^2(1-u) \dd u = \frac{4}{3}
		\] 
	\end{answer}
	
\end{example}

\begin{example}
	Let $D: -1 \leq x \leq \sin y$ and $|y| \leq \frac{\pi}{2}$ try to calculate
	\[
	I = \iint_D x \left( e^{x^2+\cos y}\sin y -1 \right) \dd x \dd y
	\] 

	\begin{answer}
		\[
			I = \iint_{D} x e^{x^2+\cos y}\sin y \dd x \dd y - \iint_{D} x \dd x \dd y
			= I_a - I_b
		\] 
		Divide the domain into two parts by a line $y=-\arcsin x$, 
		since $D_1$ is symmetric about y axis, and $x e^{x^2+\cos y}\sin y$ is odd
		about y axis, so we have first integral is zero in $D_1$, and similarly also
		get 0 in $D_2$, so we have $I_a = 0$.
		\begin{mathfig}
			I_a &= \iint_{D_1} \textcolor{red}{x} e^{x^2+\cos y}\sin y \dd x \dd y +
			\iint_{D_2} x e^{x^2+\cos y}\textcolor{red}{\sin y} \dd x \dd y \\
					&= 0 + 0 = 0 \\
			I_b &= \iint_{D_1} \textcolor{red}{x} \dd x \dd y + \iint_{D_2} x \dd x \dd y\\
					&= 0 + \iint_{D_2} x \dd x \dd y \\
					&= 2 \iint_{D_3} x \dd x \dd y

			\draw[->] (-1.5,0) -- (1.5,0) node[right] {$x$};

			\fill[blue!20, opacity=.8, domain=-pi/2:pi/2, samples=100, variable=\y]
				plot ({sin(\y r)}, {\y}) -- (-1, pi/2) -- (-1, -pi/2) -- cycle;
			% x = sin y
			\draw[blue, domain=-pi/2:pi/2, samples=100, variable=\y] plot ({sin(\y r)}, {\y});

			\fill[cyan!20, opacity=.8, domain=0:pi/2, samples=100, variable=\y]
				plot ({-cos(\y r)},pi/2-\y) -- plot ({sin(\y r)}, {\y}) -- cycle;
			% half of D_2

\section{Polar Coordinates}

\[
\iint_D f(x, y) \dd x \dd y = \iint_D f(r \cos \theta, r \sin \theta) r \dd r \dd \theta
\] 

When $f$ contains $x^2 + y^2$, we can use the polar coordinates to simplify the $\theta$
since $x^2 + y^2 = r^2$. 
When $f$ is homogeneous, that 
we recommend to use polar coordinates, so that we can eliminate the $r$ and only need to integrate $\theta$.

\subsection{Common Polar Regions}

\subsubsection{Eccentric circles}

There are two cases of common eccentric circles, both of them's center is on the
$x$ axis or $y$ axis, and pass through the origin. 
\[
	r = \pm \text{diameter} \cdot \sin \text{or} \cos \theta
\]

the derivation is simple, take x-axis as an example, the circle's equation
\[
	(x-a)^2 + y^2 = a^2 \iff x^2 + y^2 = 2ax
	\iff r^2 = 2ar\cos\theta \iff r = 2a\cos\theta
\] 

Inversely, this results gives us a way to find the graph of this kind of circle,
when we have a circle with polar equation $r = \pm 2a \cos\theta$ or $r = \pm 2a \sin\theta$, 
\begin{itemize}
	\item $\cos$ means the center is on the $x$ axis, $\sin$ means the center is on the $y$ axis.
	\item The sign of the equation indicates which side of the axis the center is located.
	\item  $a$ is the distance from the center to the origin, and also the radius of the circle.
\end{itemize}

\begin{remark}
	When given a very ugly expression of integrand function in polar coordinates
	, we can consider change it back to cartesian coordinates. Especially when
	the region is a some simple shape in cartesian coordinates, such as $r=a / \cos \theta$
	is a line  $x=a$, which is much more easier to tackle than in polar coordinates.
	Or the region is the intersection of serval shapes, and need to piecewise integrate,
	which maybe easier to integrate in cartesian coordinates.

	Besides, when the integrand function contains many trigonometric functions,
	and lots of them was multiplied with $r$, its will become simple  $x, y$ in
	 cartesian coordinates, we recommend to try change it back to cartesian coordinates.
\end{remark}

\section{Difficult To Plot Region}

Sometimes, the region is difficult to plot, we do not need to actually correctly
plot the region, we only need to find some important points, such as the intersection,
start point, end point and so on, then we only need to draw a rough sketch of the region.

\begin{example}
	Let $D:|x|\leq y, (x^2+y^2)^3 \leq y^4$, try to calculate the integral
	\[
		I = \iint_D \frac{x+y}{\sqrt{x^2+y^2}} \dd x \dd y
	\] 
	\begin{answer}
		Take special points, let $x=0$ then $y=0, 1$, let $y=0$ then $x=0$,
		let $x=y$ then $y=\pm 1/\sqrt{8}$, and notice
		that the boundary curve is symmetric about both $x$ and $y$ axis. So we
		can draw a rough sketch,
		\begin{mathfig}
			I &= 2\iint_{D_1} \frac{y}{\sqrt{x^2+y^2}} \dd x \dd y \\
				&= 2\int_{\pi/4}^{\pi/2} \dd \theta \int_0^{\sin^2\theta} r \sin\theta \dd r \\
				&= \int_{\pi/4}^{\pi/2} \sin^5\theta \dd \theta 
				= \frac{43}{120}\sqrt2
			\tikzfig[scale=2]
			%axis
			\draw[->] (-1,0) -- (1,0) node[right] {$x$};
			\draw[->] (0,-.2) -- (0,1.2) node[above] {$y$};

\section{Parameter equation bundary}

boundary as $y=y(x)$ or $x=x(y)$, directly substitute to the limit of the inner integral,
then substitute the parameter equation.

\section{Coordinate Transformation}

Let $x = x(u, v)$ and $y = y(u, v)$ be a transformation from $(u, v)$ to $(x, y)$,
then we have
\[
	\iint_D f(x, y) \dd x \dd y = \iint_{D'} f(x(u, v), y(u, v)) 
	|J| \dd u \dd v
\] 
where $J$ is the Jacobian determinant of the transformation, defined as
\[
	J = \frac{\partial(x, y)}{\partial(u, v)} =
	\begin{vmatrix}
		x_u & x_v \\
		y_u & y_v
	\end{vmatrix}
\]
\begin{remark}
	Remember to change the domain $D$ to $D'$ when changing the variables, the new
	domain $D'$ is the region in the $uv$ plane that corresponds to the original region $D$ in the $xy$ plane
\end{remark}

\begin{example}
	Let $D$ be the region bounded by the curves $xy=1, xy=2, y=x, y=4x$ in the first
	quadrant, try to calculate the integral
	\[
	I = \iint_D xy \dd x \dd y
	\] 
	\begin{answer}
		Let $u=xy, v=y/x$, then we have
		\[
			x = \sqrt{\frac{u}{v}}, y = \sqrt{uv}
		\] 
		then we have
		\[
			J = 		\begin{vmatrix}
				\frac{1}{2\sqrt{uv}} & -\frac{\sqrt{u}}{2v^{3/2}} \\
				\frac{\sqrt{v}}{2\sqrt{u}} & \frac{\sqrt{u}}{2\sqrt{v}}
			\end{vmatrix} = \frac{1}{2v}
		\] 
		The new domain $D'$ is bounded by the curves $u=1, u=2, v=1, v=4$, so we have
		\[
			I = \iint_{D'} u \left| \frac{1}{2v} \right| \dd u \dd v
			= \int_1^2 \dd u \int_1^4 \frac{u}{2v} \dd v
			= \frac{3}{2} \ln 2
		\]
		
	\end{answer}
	
\end{example}

\begin{lemma}
	Jacobian determinant has the following properties
	\[
		J(x, y) =	\frac{\partial(x, y)}{\partial(u, v)} =
		\left( \frac{\partial(u, v)}{\partial(x, y)}\right)^{-1} = 1 / J(u, v)
	\] 
\end{lemma}

\begin{example}
	Use last example's transformation, we can calculate the Jacobian determinant by
	\[
		J(u, v) = \begin{vmatrix}
			u_x & u_y \\
			v_x & v_y
		\end{vmatrix} = \begin{vmatrix}
			y & x \\
			-y/x^2 & 1/x
		\end{vmatrix} = \frac{2y}{x} = 2v = 1 / J(x, y)
	\] 
\end{example}

\section{View another variable as a constant}

Sometimes, inner integral with another variable as parameter is very confusing,
hide that it maybe just a simple integral with parameter becomes a variable.

\begin{example}
	\[
		I = \int_0^1 \dd x \int_0^{x^2} \sqrt{x^{4} - y^2} \dd y
	\] 
	\begin{answer}
		Let $t=\sqrt{x^4 - y^2}$, then $t^2 + y^2 = x^4$, which is a circle with radius $x^2$.
		The inner integral is the area of a quarter circle.
		\[
			I = \int_0^1 \frac{\pi x^4}{4} \dd x
			= \frac{\pi}{20}
		\] 
	\end{answer}
\end{example}

\subsection{Integral of Periodic Function}

When the integrand function is periodic, we can use the property of periodic function to simplify the integral.
\begin{lemma}
	Let $f(x)$ be a periodic function with period $T$, then we have
	\[
		\int_a^{a+T} f(x) \dd x = \int_0^T f(x) \dd x
	\]
	The integral of a periodic function over any interval of length equal to one period T is the same.
\end{lemma}

This is very useful when the integrand function is periodic and the limits of integration
just happen to be a(or multiple of) period, we can use this property to simplify the integral.

\begin{example}
	Let $D:0 \leq x \leq \pi, 0 \leq y \leq \pi$, try to calculate the integral
	\[
	I = \iint_D |\cos(x-y)| \dd x \dd y
	\] 
	\begin{answer}
		Notice that $|\cos(x-y)|$ is a periodic function with period $\pi$ regarding
		$x$ or $y$ when considering the other variable as a constant, so we have
		\[
			I = \int_0^{\pi} \dd x \int_0^{\pi} |\cos(x-y)| \dd y
			= \int_0^{\pi} \dd x \int_0^{\pi} |\cos y| \dd y
			= 2\pi
		\] 
		Which is much easier than dividing the integral into several parts(to tackle the
		absolute value) and then integrate.
	\end{answer}
\end{example}

\begin{example}
	Let $D:0 \leq x \leq y \leq 2\pi$, try to calculate the integral
	\[
	I = \iint_D |\sin(x-y)| \dd x \dd y
	\] 
	\begin{answer}
		Notice that $|\sin(x-y)|$ is symmetric about $x=y$, so we have
		\[
			I = \frac{1}{2} \int_0^{2\pi} \dd x \int_0^{2\pi} |\sin(x-y)| \dd y
			= \frac{1}{2}\cdot 2\pi\cdot 4 = 4\pi
		\] 
		
	\end{answer}	
\end{example}

\section{Change order of Iterated Integral}

When meeting a double integral with some obviously unable integrand function,
we always consider changing the order of integration, which may make the integral easier to calculate.
Here are some common cases of unable functions,
\begin{itemize}
	\item $\exp(\pm x^2), \exp(1 / x), 1 / \ln x$

	\item $\sqrt{x^{3}+1}, 1 / \sqrt{x^{3}+1}$
\end{itemize}

or very difficult to integrate functions
\begin{itemize}
	\item $\sqrt{a\pm x^2}$
\end{itemize}

\subsection{Change of order in polar coordinates}

When changing the order of integration in polar coordinates, in order to determine
the region of integration, there are two common methods, one is to directly
analyze the region in polar coordinates draw in xOy plane.

Then to get the limits, we fix $r$ draw a circle with radius $r$, then find the
intersection points with the boundary of the region, so that find the limits of
 $\theta$ for the fixed $r$.

Another is to consider
$r, \theta$ as perpendicular coordinates, then tackle as the rectangular coordinates
which is similar to the $xOy$ plane. 

\begin{example}
	\[
		I = \int_0^{2\pi} \dd \theta \int_{\theta /2}^{\pi} \theta^2 e^{r^2} \dd r
	\] 

\begin{answer}
	\[
		\begin{aligned}
			I &= \int_0^{\pi} \dd r \int_0^{2r} \theta^2 e^{r^2} \dd \theta
			 = \int_0^{\pi} e^{r^2} \dd r \frac{1}{3} (2r)^3
			 = \frac{8}{3} \int_0^{\pi} r^3 e^{r^2} \dd r
				  = \frac{4}{3} e^{\pi^2} (\pi^2 - 1) + \frac{4}{3}
			\end{aligned}
	\]
