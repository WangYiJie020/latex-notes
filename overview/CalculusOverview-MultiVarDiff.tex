\subsection{Multiple Limits}

Definition:
\[
    \lmxyoo f(x,y) = L
\]
if for every $\epsilon > 0$, there exists $\delta > 0$ such that when
\[
    0 < \sqrt{(x-x_0)^2 + (y-y_0)^2} < \delta,
\]
we have $|f(x,y) - L| < \epsilon$.

Usually, to prove that limits do not exist,
we choose different paths approaching $(x_0,y_0)$ 
and show that the limits along those paths are different.

\begin{theorem}
\label{thm:xayb_over_x2y2_lim}
For $\dfrac{x^{\alpha}y^{\beta}}{x^2+y^2}$ we use polar coordinates to
determine the limit as $r \to 0$. ($\alpha, \beta>0$)
\[
	\lmxyoo \frac{x^{\alpha}y^{\beta}}{x^2+y^2}
	= \lim_{r \to 0^{+}} 
	\frac{r^{\alpha+\beta} \cos^{\alpha} \theta \sin^{\beta} \theta }{r^2}
	= \lim_{r \to 0^{+}} r^{\alpha+\beta-2} \cos^{\alpha} \theta \sin^{\beta} \theta
	= \begin{cases}
		0 & \text{if } \alpha + \beta > 2 \\
		\text{does not exist} & \text{if } \alpha + \beta \leq 2
	\end{cases}
	\]
\end{theorem}

\begin{example}
	Also useful with similar form:
	\[
		\lmxyoo \frac{x^2y^2}{(x^2+y^2)^{\frac{3}{2}}}
		= \lmxyoo \left( \frac{x^{\frac{4}{3}}y^{\frac{4}{3}}}{x^2+y^2} \right)^{\frac{3}{2}}
		= 0
	\] 
\end{example}

\begin{example}
	By substituting variables, we can apply the same method to more cases.
	\[
		\lmxyoo \frac{x^{6}y^{7}}{x^{6}+y^{8}}
		\xlongequal[v=y^4]{u=x^3}
		\jlm{u}{v}{0}{0^{+}} \frac{u^2v^{\frac{7}{4}}}{u^2+v^2}
	=0
	\] 
\end{example}

But be careful that after converting to polar coordinates, \textbf{limits along
all fixed angles may exist without implying that the original limit exists.}

\begin{example}
	\[
		f(x, y) = \frac{x^2y}{x^4+y^2}
	\] 
In polar coordinates, we have
\[
	f(r, \theta) = \frac{r^2 \cos^2 \theta \sin \theta}{r^4 \cos^4 \theta + r^4 \sin^2 \theta}
	= \frac{r\cos^2 \theta \sin \theta}{r^2 \cos^4 \theta + \sin^2 \theta}
\]
For any fixed $\theta$, we have $\lim_{r \to 0} f(r, \theta) = 0$.
However, if we choose the path $y = x^2$, we have
\[
	\lim_{x \to 0} f(x, x^2) = \lim_{x \to 0} \frac{x^{4}}{2x^{4}} = \frac{1}{2}
\]

The key point is that when we approach $(0,0)$ along the path $y = x^2$, the
angle $\theta$ is not fixed, but approaches $0$ as $x \to 0$. Actually, we
cannot conclude that the limit $\lim_{(x,y) \to (0,0)} f(x,y)$ exists.
	
\end{example}

\subsection{Differentiability}

Definition:
$f$ is differentiable at $(x_0,y_0)$ if there exist $A$ and $B$ such that
\[
	\Delta f = A \Delta x + B \Delta y + o(\sqrt{(\Delta x)^2 + (\Delta y)^2})
\] 
Or equivalently, we can write
\[
	\jlm{\Delta x}{\Delta y}{0}{0}
	\frac{\Delta f - A \Delta x - B \Delta y}{\sqrt{(\Delta x)^2 + (\Delta y)^2}} = 0
\]

When $f$ is differentiable at $(x_0,y_0)$, we have

\[
\dd f = A \dd x + B \dd y = f_x(x_0,y_0) \dd x + f_y(x_0,y_0) \dd y
\]

\begin{theorem}
	If $f(x, y)$ is continuous at $(x_0, y_0)$ and
	 \[
		 \lmxy{x_0}{y_0} \frac{f(x,y) - Ax - By - C}{\sqrt{(x-x_0)^2 + (y-y_0)^2}} = 0
	\] 
	then $f$ is differentiable at $(x_0, y_0)$ and $A = f_x(x_0, y_0)$, $B = f_y(x_0, y_0)$.
	
\end{theorem}

\begin{proof}
Let
\[
r=\sqrt{(x-x_0)^2+(y-y_0)^2}.
\]
since \(r\to 0\) as \((x,y)\to (x_0,y_0)\), we have
\[
f(x,y)-Ax-By-C\to 0.
\]
using the continuity of \(f\) at \((x_0,y_0)\), it follows that
\[
f(x_0,y_0)-Ax_0-By_0-C=0.
\]
thus
\[
C=f(x_0,y_0)-Ax_0-By_0.
\]
therefore the assumed limit becomes
the definition of differentiability of \(f\) at \((x_0,y_0)\).
\[
\lim_{(x,y)\to (x_0,y_0)}
\frac{
f(x,y)-f(x_0,y_0)-A(x-x_0)-B(y-y_0)
}{
\sqrt{(x-x_0)^2+(y-y_0)^2}
}
=0.
\]

To identify \(A\) and \(B\). Taking \(y=y_0\), we get
\[
\lim_{x\to x_0}
\frac{
f(x,y_0)-f(x_0,y_0)-A(x-x_0)
}{
|x-x_0|
}
=0.
\]
hence
\[
\frac{f(x,y_0)-f(x_0,y_0)}{x-x_0}\to A,
\]
so
\[
f_x(x_0,y_0)=A.
\]
similarly, taking \(x=x_0\), we obtain
\[
f_y(x_0,y_0)=B.
\]
\end{proof}

\begin{example}
	$f(x, y)$ is continuous at $(0, 0)$ and
	\[
		\lmxyoo \frac{f(x,y) + 3x - 4y}{x^2 + y^2} = 2
	\]
	try to find $f_x(0, 0)$ and $f_y(0, 0)$.
	\begin{answer}
		The limit can be rewritten as
		\[
			f + 3x - 4y = 2(x^2 + y^2) + o(x^2+y^2)
		\]
		dividing both sides by $\sqrt{x^2 + y^2}$, we have
		\[
			\frac{f + 3x - 4y}{\sqrt{x^2 + y^2}} = 2\sqrt{x^2 + y^2} + o(\sqrt{x^2 + y^2}) 
		\] 
		which implies
		\[
			\lmxyoo \frac{f + 3x - 4y}{\sqrt{x^2 + y^2}} = 0
		\] 
		gives that $f$ is differentiable at $(0, 0)$ and $f_x(0, 0) = -3$, $f_y(0, 0) = 4$.
	\end{answer}
\end{example}

\begin{theorem}
	Let
	\[
	f(x,y) = \begin{cases}
		\dfrac{x^{\alpha}y^{\beta}}{x^2+y^2} & x^2+y^2 \neq 0 \\
		0 & x^2+y^2 = 0
	\end{cases}
	\]
	where $\alpha, \beta > 0$. Then
		\begin{itemize}
			\item $f(x,y)$ is continuous at $(0,0)$ if and only if $\alpha + \beta > 2$.
			\item $f(x,y)$ is differentiable at $(0,0)$ if and only if $\alpha + \beta > 3$.
		\end{itemize}
\end{theorem}

\begin{proof}
	For continuity, we have $f(0,0) = 0$. By \ref{thm:xayb_over_x2y2_lim}, we have
		\[
			\lmxyoo f(x,y) = \begin{cases}
				0 & \text{if } \alpha + \beta > 2 \\
				\text{does not exist} & \text{if } \alpha + \beta \leq 2
			\end{cases}
		\] 
	Gives that $f(x,y)$ is continuous at $(0,0)$ if and only if $\alpha + \beta > 2$.
		
		For differentiability, we have $f_x(0,0) = f_y(0,0) = 0$. Use the definition
		of differentiability,
		\begin{align*}
			& \lmxyoo \frac{f(x,y) - f(0,0) - f_x(0,0) x - f_y(0,0) y}{\sqrt{x^2 + y^2}} \\
			=& \lmxyoo \frac{x^{\alpha}y^{\beta}}{(x^2+y^2)^{3/2}}\\
			=& \lmxyoo \left( \frac{x^{\frac{2\alpha}{3}}y^{\frac{2\beta}{3}}}{x^2+y^2} \right)^{\frac{3}{2}}\\
		\end{align*}

	By \ref{thm:xayb_over_x2y2_lim}, the limit exists and equals $0$ if and only if
	\[
		\frac{2\alpha}{3} + \frac{2\beta}{3} > 2 \iff \alpha + \beta > 3
	\] 

\end{proof}

\subsubsection{Partial Derivatives of Piecewise-Defined Functions}

For a piecewise-defined function, the main question is whether moving in the
differentiation direction stays inside the same region.

\begin{itemize}
    \item If the differentiation direction remains inside the same region, 
			we can differentiate the corresponding formula directly and then substitute the point.
    \item If the differentiation direction crosses a region boundary, 
			must use the definition of the partial derivative.
\end{itemize}

\begin{example}
Let
\[
f(x,y)=
\begin{cases}
x+y, & x>0,\\
y, & x=0,\ y\neq 0,\\
2x+y, & x<0.
\end{cases}
\]

At $(0,1)$, the $y$-direction stays on the line $x=0$, so
\[
f_y(0,1)=1.
\]

For the partial derivative with respect to $x$,
\[
\frac{f(h,1)-f(0,1)}{h}
=
\begin{cases}
1, & h>0,\\
2, & h<0.
\end{cases}
\]

The two one-sided limits are different, so $ f_x(0,1) $ does not exist.
While $\partial_x f_{\text{at } x=0, y\neq 0} = 0$ exists.
\end{example}

\subsubsection{Mixed Partial Derivatives}

When work with mixed partial derivatives at a point, we can't substitute the point
into the formula directly to get first order partial derivatives.

\begin{example}
	Let $f(x, y)=\ln |x+y|-\sin(xy)$, find $f_{xy}(1,\pi)$.

	\begin{answer}
		\[
			f_x(x, y) = \frac{1}{x+y} - y\cos(xy)
		\]
		subtituting $x=1$ gives  $f_x(1, y) = {1}/{(1+y)} - y\cos(y)$, then
		\[
			f_{xy}(1, y) = -\frac{1}{(1+y)^2} - \cos(y) + y\sin(y)
		\]
		substituting $y=\pi$ gives $f_{xy}(1, \pi) = -1 / (1+\pi)^2 -1$.

		
	\end{answer}
	
	\begin{remark}
		A mistake is substituting $y=\pi$ when computing $f_x$ which gives
		\[
			f_x(x, \pi) = \frac{1}{x+\pi} - \pi\cos(\pi x)
		\]
	\end{remark}

\end{example}


\subsection{Partial differentiation of composite functions}

\begin{example}

Let
\[
w=xy-z,
\]
where \(z=z(x,y)\) has continuous second-order partial derivatives and satisfies
\[
\frac{\partial^2 z}{\partial x^2}
+2\frac{\partial^2 z}{\partial x\partial y}
+\frac{\partial^2 z}{\partial y^2}
=0
\]
introduce the transformation
\[
u=x+y,\qquad v=x-y
\]
try to find
\[
\frac{\partial^2 w}{\partial u^2}
\]

\begin{answer}

By the chain rule,
\[
\partial_x = \partial_u u_x + \partial_v v_x
= \partial_u + \partial_v
\]
\[
\partial_y = \partial_u u_y + \partial_v v_y
= \partial_u - \partial_v
\]
hence,
\begin{align*}
z_{xx} &=
\left(\partial_u + \partial_v\right)^2 z
= z_{uu}+2z_{uv}+z_{vv}\\
z_{xy}
&=
\left( \partial_u + \partial_v \right)
\left( \partial_u - \partial_v \right)z
=
z_{uu}-z_{vv}\\
z_{yy}
&= \left( \partial_u - \partial_v \right)^2 z
=
z_{uu}-2z_{uv}+z_{vv}
\end{align*}
therefore,
\[
z_{xx}+2z_{xy}+z_{yy}
=
4z_{uu}
\]
gives
\[
z_{uu}=0
\]
we have
\[
x=\frac{u+v}{2},\qquad y=\frac{u-v}{2} ,\qquad xy = \frac{u^2-v^2}{4}
\]
so
\[
w
=
\frac{u^2-v^2}{4}-z
\]
differentiating twice with respect to \(u\),
\[
\frac{\partial w}{\partial u}
=
\frac{u}{2}-z_u
\]
and hence
\[
\frac{\partial^2 w}{\partial u^2}
= \frac12-z_{uu} = 0
\]
	
\end{answer}
\end{example}

\subsubsection{Multivariable Chain Rule}

\begin{theorem}
	\label{thm:multivariable_chain_rule}
	Let $\boldsymbol{x}=(x_1, x_2, \ldots, x_n)^T$ and $\boldsymbol{y}=(y_1, y_2, \ldots, y_m)^T$, then
	\[
		\nabla_{\boldsymbol{x}} = \boldsymbol{J}^T \nabla_{\boldsymbol{y}}
	\] 
	where $\boldsymbol{J}$ is the Jacobian matrix of the transformation $\boldsymbol{y} = \boldsymbol{y}(\boldsymbol{x})$.
	\[
		J_{ij} = \frac{\partial y_i}{\partial x_j}
	\] 
	and
	\[
		\nabla_{\boldsymbol{x}} = \left(\partial_{x_1}, \partial_{x_2}, \ldots, \partial_{x_n}\right)^T,\quad
	\] 
\end{theorem}
\begin{proof}
	For a function $u(\boldsymbol{y})$, we have
	\[
		\dd u = \sum\frac{\partial u}{\partial y_i} \dd y_i = (\nabla_{\boldsymbol{y}}u)^T \dd \boldsymbol{y}
		= (\nabla_{\boldsymbol{y}}u)^T \boldsymbol{J} \dd \boldsymbol{x}
	\] 
	comparing with
	\[
		\dd u = \frac{\partial u}{\partial x^j} \dd x^j = (\nabla_{\boldsymbol{x}}u)^T \dd \boldsymbol{x}
	\] 
	gives
	\[
		\nabla_{\boldsymbol{x}}u = \boldsymbol{J}^T \nabla_{\boldsymbol{y}}u
	\]
\end{proof}

\subsection{Simplifying partial differential equations by a change of variables}

\begin{lemma}[Frechet derivative]
	\label{lem:Frechet_derivative}
	\[
	D_{\boldsymbol{x}} \boldsymbol{f} = \frac{\partial (f_1, f_2, \ldots, f_m)}{\partial (x_1, x_2, \ldots, x_n)}
		= \frac{\partial f_i}{\partial x_j}
	\] 
	satisfies the chain rule
	\[
		D_{\boldsymbol{x}} (\boldsymbol{u} \compose \boldsymbol{f}) = 
		(D_{\boldsymbol{u}} \boldsymbol{f}) D_{\boldsymbol{x}} \boldsymbol{u}
	\] 
\end{lemma}

\begin{theorem}
	\label{thm:order2_Hessian_matrix}
	Second-order Hessian matrix satisfies
	\[
		\boldsymbol{H}_{xy} = \boldsymbol{J}^T \boldsymbol{H}_{\xi\eta} \boldsymbol{J}
		+ u_{\xi} \boldsymbol{H}_{xy}(\xi) + u_{\eta} \boldsymbol{H}_{xy}(\eta)
	\] 
	where
	\[
		\boldsymbol{H}_{xy} = \boldsymbol{H}_{xy}(u) = \begin{bmatrix} u_{xx} & u_{xy}\\ u_{yx} & u_{yy} \end{bmatrix},\quad
	\]
	and the Jacobian matrix
	\[
		\boldsymbol{J} = \begin{bmatrix} \xi_x & \xi_y\\ \eta_x & \eta_y \end{bmatrix}
	\] 
	
\end{theorem}
\begin{proof}
Using the chain rule and the Jacobian matrix, we have
\begin{equation} \label{eq:first_order_HSM_chain_rule}
	\begin{bmatrix} u_x\\ u_y \end{bmatrix}
	=
	\boldsymbol{J}^T
	\begin{bmatrix} u_{\xi}\\ u_{\eta} \end{bmatrix}
\end{equation}
again by the chain rule
\[
	\begin{bmatrix} u_{\xi x} \\ u_{\xi y} \end{bmatrix}
	= 
	\boldsymbol{J}^T \begin{bmatrix} u_{\xi\xi}\\ u_{\xi\eta} \end{bmatrix}
	, \quad
	\begin{bmatrix} u_{\eta x} \\ u_{\eta y} \end{bmatrix}
	= 
	\boldsymbol{J}^T \begin{bmatrix} u_{\eta\xi}\\ u_{\eta\eta} \end{bmatrix}
\] 
	transpose above two equations and combine them gives
	\begin{equation} \label{eq:second_order_HSM_chain_rule}
		\begin{bmatrix} u_{\xi x} & u_{\xi y} \\ u_{\eta x} & u_{\eta y} \end{bmatrix}
		=
		\begin{bmatrix}
			u_{\xi\xi} & u_{\xi\eta}\\
			u_{\eta\xi} & u_{\eta\eta}
		\end{bmatrix}
		\boldsymbol{J}
	\end{equation}
	take derivative of \eqref{eq:first_order_HSM_chain_rule} with respect to $x$ and $y$ gives
	\[
		\begin{bmatrix} u_{xx} & u_{xy}\\ u_{yx} & u_{yy} \end{bmatrix}
		=
		\begin{bmatrix}
		\boldsymbol{J}^T_x
		\begin{bmatrix} u_{\xi}\\ u_{\eta} \end{bmatrix}
		&
		\boldsymbol{J}^T_y
		\begin{bmatrix} u_{\xi}\\ u_{\eta} \end{bmatrix}
		\end{bmatrix}
		+
		\boldsymbol{J}^T
		\begin{bmatrix} u_{\xi x} & u_{\xi y}\\ u_{\eta x} & u_{\eta y} \end{bmatrix}
	\] 
	where \[
		\boldsymbol{J}^T_x = \begin{bmatrix} \xi_{xx} & \eta_{xx}\\ \xi_{yx} & \eta_{yx} \end{bmatrix}
		\quad
		\boldsymbol{J}^T_y = \begin{bmatrix} \xi_{xy} & \eta_{xy}\\ \xi_{yy} & \eta_{yy} \end{bmatrix}
	\] 
	opening the matrix multiplication gives
	\[
	\boldsymbol{J}^T_x
	\begin{bmatrix}
	u_\xi\\
	u_\eta
	\end{bmatrix}
	=
	\begin{bmatrix}
	u_\xi \xi_{xx} + u_\eta \eta_{xx}\\
	u_\xi \xi_{yx} + u_\eta \eta_{yx}
	\end{bmatrix},
	\]
	and
	\[
	\boldsymbol{J}^T_y
	\begin{bmatrix}
	u_\xi\\
	u_\eta
	\end{bmatrix}
	=
	\begin{bmatrix}
	u_\xi \xi_{xy} + u_\eta \eta_{xy}\\
	u_\xi \xi_{yy} + u_\eta \eta_{yy}
	\end{bmatrix}
\]
gives
\[
\begin{aligned}
\begin{bmatrix}
\boldsymbol{J}^T_x \begin{bmatrix} u_\xi\\ u_\eta \end{bmatrix}
&
\boldsymbol{J}^T_y \begin{bmatrix} u_\xi\\ u_\eta \end{bmatrix}
\end{bmatrix}
&=
\begin{bmatrix}
u_\xi \xi_{xx} + u_\eta \eta_{xx}
&
u_\xi \xi_{xy} + u_\eta \eta_{xy}
\\
u_\xi \xi_{yx} + u_\eta \eta_{yx}
&
u_\xi \xi_{yy} + u_\eta \eta_{yy}
\end{bmatrix}
\\
&=
u_\xi
\begin{bmatrix}
\xi_{xx} & \xi_{xy}\\
\xi_{yx} & \xi_{yy}
\end{bmatrix}
+
u_\eta
\begin{bmatrix}
\eta_{xx} & \eta_{xy}\\
\eta_{yx} & \eta_{yy}
\end{bmatrix}
\\
&=
u_\xi \boldsymbol{H}_{xy}(\xi)
+
u_\eta \boldsymbol{H}_{xy}(\eta).
\end{aligned}
\]
for the second term, substituting \eqref{eq:second_order_HSM_chain_rule} gives
\[
	\boldsymbol{J}^T
	\begin{bmatrix} u_{\xi x} & u_{\xi y}\\ u_{\eta x} & u_{\eta y} \end{bmatrix}
	=
	\boldsymbol{J}^T \boldsymbol{H}_{\xi\eta} \boldsymbol{J}
\]
substituting the above two equations into the previous equation gives the desired result
\end{proof}

\begin{remark}
	A more convenient way to prove the theorem is to use tensor notation, or 
	consider the bilinear form associated with the second differential.
	But those exceeds the scope of this notebook.
\end{remark}

\begin{proof}
	By \ref{thm:multivariable_chain_rule},  we provide another fancy method, since
	\begin{equation}
		\nabla _{xy}u = \boldsymbol{J}^T \nabla_{\xi\eta}u
	\end{equation}
	we have
	\[
		\boldsymbol{H}_{xy} = D_{xy}(\nabla_{xy} u)
	\] 
	that is
	\[
		\boldsymbol{H}_{xy}
		= D_{xy}(\boldsymbol{J}^T )\nabla_{\xi\eta} u + \boldsymbol{J}^T D_{xy}(\nabla_{\xi\eta} u)
	\] 
	for the second term, by \ref{lem:Frechet_derivative},
	\[
		D_{xy}(\nabla_{\xi\eta} u) = D_{\xi\eta}(\nabla_{\xi\eta} u) D_{xy}(\xi, \eta)
		= \boldsymbol{H}_{\xi\eta} \boldsymbol{J}
	\] 
	for the first term, we have
	\[
		D_{xy}(\boldsymbol{J}^T) = D_{xy} (\nabla_{xy} \xi, \nabla_{xy} \eta)
		= (\boldsymbol{H}_{xy}(\xi), \boldsymbol{H}_{xy}(\eta))
	\]
	multiplying by $\nabla_{\xi\eta} u$ and collecting the terms gives
	\[
		\boldsymbol{H}_{xy} = \boldsymbol{J}^T \boldsymbol{H}_{\xi\eta} \boldsymbol{J}
		+ u_{\xi} \boldsymbol{H}_{xy}(\xi) + u_{\eta} \boldsymbol{H}_{xy}(\eta)
	\] 
\end{proof}

\begin{theorem}[Inference of linear transformation]
	\label{thm:linear_transformation_Hessian_matrix}
	When $\xi$ and $\eta$ are linear functions of $x$ and $y$
\[
\begin{bmatrix} \xi\\ \eta \end{bmatrix}
= \boldsymbol{A} \begin{bmatrix} x\\ y \end{bmatrix}
= \begin{bmatrix} a_{11} & a_{12}\\ a_{21} & a_{22} \end{bmatrix}
\begin{bmatrix} x\\ y \end{bmatrix}
=
\begin{bmatrix} \xi_x & \xi_y\\ \eta_x & \eta_y \end{bmatrix}
\begin{bmatrix} x\\ y \end{bmatrix}
= \boldsymbol{J} \begin{bmatrix} x\\ y \end{bmatrix}
\]
we have
\[
	\boldsymbol{H}_{xy} = \boldsymbol{J}^T \boldsymbol{H}_{\xi\eta} \boldsymbol{J}
	= A^T \boldsymbol{H}_{\xi\eta} A
\] 
\end{theorem}
\begin{proof}
	Since $\xi$ and $\eta$ are linear functions of $x$ and $y$, all second-order
	derivatives are zero, apply \ref{thm:order2_Hessian_matrix} gives the result.
\end{proof}

\begin{theorem}
For a second-order partial differential equation of the form
\[
	p\frac{\partial^2 u}{\partial x^2}
	+ q\frac{\partial^2 u}{\partial x\partial y}
	+ r\frac{\partial^2 u}{\partial y^2}
	=0
\] 
Intrduce change of variables
\[
\begin{bmatrix} \xi\\ \eta \end{bmatrix}
= \boldsymbol{A} \begin{bmatrix} x\\ y \end{bmatrix}
= \begin{bmatrix} a & b\\ c & d \end{bmatrix}
\begin{bmatrix} x\\ y \end{bmatrix}
\] 
and notation
\[
	P = u_{\xi\xi},\quad Q = u_{\xi\eta},\quad R = u_{\eta\eta}
\] 
then
\[

	\] 
	
\end{proof}

\subsection{Implicit Differentiation Formula}

Suppose that
\[
F(x,y,z)=0
\]
defines \(z\) implicitly as a function of \(x\) and \(y\), say
\[
z=z(x,y).
\]
If \(F\) is sufficiently smooth, then differentiating with respect to \(x\), while keeping \(y\) fixed, gives
\[
F_x+F_z z_x=0.
\]
Hence
\[
z_x=\frac{\partial z}{\partial x}
=
-\frac{F_x}{F_z}.
\]

A standard sufficient condition is the following. Let
\[
F(x_0,y_0,z_0)=0.
\]
If \(F\in C^1\) in a neighborhood of \((x_0,y_0,z_0)\) ($F$ has continuous first-order partial derivatives), and
\[
F_z(x_0,y_0,z_0)\neq 0,
\]
then, near \((x_0,y_0,z_0)\), the equation
\[
F(x,y,z)=0
\]
uniquely defines \(z\) as a \(C^1\) function of \(x\) and \(y\). Moreover,
\[
\left.\frac{\partial z}{\partial x}\right|_y
=
-\frac{F_x}{F_z}.
\]

The condition \(F_z\neq 0\) alone is not enough if \(F\) is not sufficiently differentiable.

\begin{example}[A Counterexample]	

It is enough to consider the two-variable case
\[
F(x,z)=0,
\]
which corresponds to the three-variable case where \(F\) is independent of \(y\).
\[
F(x,z)=
\begin{cases}
(z-x)\left(\dfrac{z^2}{x^2+z^2}+x^2\right),
& (x,z)\neq (0,0),\\[1em]
0,
& (x,z)=(0,0).
\end{cases}
\]

We examine the point \((0,0)\). First,
\[
F_z(0,0)
=
\lim_{h\to 0}\frac{F(0,h)-F(0,0)}{h}.
\]
since
\[
F(0,h)=h,
\]
we get
\[
F_z(0,0)=1\neq 0.
\]
next,
\[
F_x(0,0)
=
\lim_{h\to 0}\frac{F(h,0)-F(0,0)}{h}.
\]
since
\[
F(h,0)=(-h)h^2=-h^3,
\]
we get
\[
F_x(0,0)
=
\lim_{h\to 0}(-h^2)=0.
\]

Therefore, if one formally applies the implicit differentiation formula at the origin, one obtains
\[
-\frac{F_x(0,0)}{F_z(0,0)}
=
-\frac{0}{1}
=
0.
\]

However, for \((x,z)\neq (0,0)\), the factor
\[
\dfrac{z^2}{x^2+z^2}+x^2
\]
is strictly positive. Hence
\[
F(x,z)=0
\]
is equivalent to
\[
z-x=0.
\]
thus the implicit function is simply
\[
z=x.
\]
therefore,
\[
z_x(0)=1.
\]
So the true derivative is
\[
z_x(0)=1,
\]
while the formula gives
\[
-\frac{F_x(0,0)}{F_z(0,0)}=0.
\]

\end{example}

\begin{remark} The failure occurs because \(F\) is not differentiable
	at the origin. Although the partial derivatives
\[
F_x(0,0),\qquad F_z(0,0)
\]
exist and \(F_z(0,0)\neq 0\), the hypotheses of the implicit function theorem are not satisfied.
In particular, \(F_x\) is not continuous at the origin. Along the line \(z=x\),
\[
F_x(x,x)\to -\frac12
\qquad \text{as } x\to 0,
\]
whereas
\[
F_x(0,0)=0.
\]

Thus the existence of partial derivatives at a point is not sufficient. The usual formula
\[
z_x=-\frac{F_x}{F_z}
\]
relies on the differentiability of \(F\), or more commonly on the stronger assumption \(F\in C^1\),
so that the chain rule and the implicit function theorem apply.
\textbf{The chain rule requires differentiability.}

\end{remark}

\begin{remark}
	When tackling problems involving those forms below
	\[
		\frac{x^2}{x^2+y^2},\quad
		\frac{xz}{x^2+y^2+z^2},\quad
		|x|,
		\sqrt{x^2+y^2},
	\] 
	or picewise-defined functions, should be careful about the differentiability
	and don't directly apply the $-F_x/F_z$ formula.
	
\end{remark}

\subsection{Find the potential function}
Given a total differential
\[
\dd z = P(x,y) \dd x + Q(x,y) \dd y
\] 
to find the potential function $z(x,y)$, we can first integrate with respect to $x$ 
\[
\int P(x,y) \dd x = f(x,y) + C(y)
\] 
then differentiate with respect to $y$ and compare with $Q(x,y)$ to find $C(y)$.

\begin{example}
	Try to find the potential function of the total differential
	\[
		\dd f = \frac{x \dd y - y \dd x}{x^2+y^2}
	\] 
	\begin{answer}
		Integrate with respect to $x$ gives
		\[
			f(x,y) = \int -\frac{y}{x^2+y^2} \dd x + C(y)
			= -\arctan\frac{x}{y} + C(y)
		\] 
		differentiate with respect to $y$ gives
		\[
			f_y(x,y) = \frac{x}{x^2+y^2} + C'(y)
		\] 
		comparing with $Q(x,y) = {x}/{(x^2+y^2)}$ gives $C'(y)=0$, hence
		\[
			C(y) = C
		\]
		therefore the potential function is
		\[
			f(x,y) = -\arctan\frac{x}{y} + C
		\]
	\end{answer}
\end{example}

\section{Extreme points of multivariable functions}

\subsection{Unconstrained Extreme}

\begin{lemma}[second-order Taylor expansion]
	\label{lem:second_order_Taylor_expansion}
	Let $f(x,y)$ have continuous second-order partial derivatives in a neighborhood of $(x_0,y_0)$, 
	\[
		\Delta f = \nabla f \cdot \boldsymbol{d}
		+ \frac12 \boldsymbol{d}^T \boldsymbol{H} \boldsymbol{d}
		+ o(\boldsymbol{d}^2)
	\] 
	where $\Delta f = f(x_0+h,y_0+k)-f(x_0,y_0)$, $\boldsymbol{d} = (h,k)^T$ and 
	$\boldsymbol{H}$ is the Hessian matrix.
	
\end{lemma}

\begin{theorem}

A necessary condition for a function $f(x,y)$ to have an extreme point at
 $(x_0,y_0)$ is that the first-order partial derivatives vanish at that point
\[
f_x(x_0,y_0) = f_y(x_0,y_0)=0
\] 
let
\[
	A = f_{xx}(x_0,y_0),\quad
	B = f_{xy}(x_0,y_0),\quad
	C = f_{yy}(x_0,y_0)
\] 
and
\[
	D = AC-B^2
\] 
then the sufficient condition for an extreme point is
\begin{itemize}
	\item If $D>0$ and $A>0$, then $(x_0,y_0)$ is a local minimum point.
	\item If $D>0$ and $A<0$, then $(x_0,y_0)$ is a local maximum point.
	\item If $D<0$, then $(x_0,y_0)$ is not an extreme point.
	\item If $D=0$, then the test is inconclusive.
\end{itemize}
	
\end{theorem}

\begin{proof}
	By \ref{lem:second_order_Taylor_expansion}, stationary point requires
	\[
	\nabla f = 0
	\] 
	determine whether the stationary point is a local maximum, local minimum or
	saddle point is equivalent to determine whether the quadratic form
	\[
	\boldsymbol{d}^T \boldsymbol{H} \boldsymbol{d}
	\]
	is positive definite, negative definite or indefinite. By the Sylvester's criterion
	here we have
	\[
		D_1 = A,\quad D_2 = |\boldsymbol{H}| = AC-B^2
	\] 
	\begin{itemize}
		\item positive definite (local minimum) requires $D_1>0$ and $D_2>0$
		\item negative definite (local maximum) requires $D_1<0$ and $D_2>0$
		\item indefinite (saddle point) when $D_2<0$
		\item inconclusive when $D_2=0$
	\end{itemize}
\end{proof}

When we meet inconclusive case, we usually consider 2 aspects
\begin{itemize}
	\item Try to proof it is extreme point by using the sign-preserving property
	\item Try to find a counterexample by taking a specific path from different directions
\end{itemize}
\begin{example}
	Try to find the extreme points of the function
	\[
	f(x,y) = x^4+y^4 - (x+y)^2
	\]
	\begin{answer}
		It's easy to find the stationary points are $(0,0)$ , $(1,1)$ and $(-1,-1)$
		and
		\[
		A = 12x^2-2,\quad B=-2,\quad C=12y^2-2
		\] 
		shows $f(1,1)=f(-1,-1)=-2$ are local maximum points, and $f(0,0)=0$ is inconclusive.
		To determine whether $(0,0)$ is an extreme point, we can consider the path $y=0$ and $y=-x$,
		for sufficiently small $x$, we have
		\[
		f(x,0) = x^4-x^2<0,\quad f(x,-x) = 2x^4>0
		\] 
		hence $(0,0)$ is not an extreme point.
	\end{answer}
\end{example}

\subsubsection{Lagrange Multiplier Method}

To find the extreme points of a function $f(x,y)$ subject to a constraint $g(x,y)=0$, 
we can use the Lagrange multiplier method. The necessary condition for an extreme
point is that there exists a scalar $\lambda$ such that
\[
\nabla f = \lambda \nabla g
\] 
and satisfying the constraint $g(x,y)=0$. Or written in explicit form and change the 
sign of $\lambda$ 
\[
	\begin{cases}
		L_x = f_x + \lambda g_x = 0\\
		L_y = f_y + \lambda g_y = 0\\
		g(x,y) = 0
	\end{cases}
\] 

When $f$ is complicated, it is not easy to solve or even write down the equations.
A trick is to consider another function $F$ has equvalent extreme points with $f$
such as
\begin{itemize}
	\item $\sqrt{u}\rightarrow u$
	\item $e^u\rightarrow u$
	\item $\ln u\rightarrow u$
	\item $|u|\rightarrow u^2$
\end{itemize}

\begin{example}
	Try to find the extreme points of the function
	\[
	f(a, b) = (b-a)^3
\]
	subject to the constraint
	\[
	g(a, b) = a^2 + b^2 - 1 = 0
	\] 
	\begin{answer}
		We can consider the equivalent function
		\[
		F(a, b) = b-a
		\] 
		by Lagrange multiplier method, we have
		\[
		\begin{cases}
			L_a = -1 + 2\lambda a = 0\\
			L_b = 1 + 2\lambda b = 0\\
			g(a, b) = a^2 + b^2 - 1 = 0
		\end{cases}
	\]
	\end{answer}
\end{example}

\subsubsection{Inequality Method}

\begin{theorem}[Cauchy-Schwarz Inequality]
	\label{thm:Cauchy-Schwarz_Inequality}
	For any vectors $\boldsymbol{u}, \boldsymbol{v} \in \mathbb{R}^n$, we have
	\[
		|\boldsymbol{u}\cdot \boldsymbol{v}| \leq |\boldsymbol{u}| |\boldsymbol{v}|
	\] 
	with equality if and only if $\boldsymbol{u}$ and $\boldsymbol{v}$ are linearly dependent.
	We usually use the following form
	\[
		\left(\sum_{i=1}^n a_i b_i\right)^2 \leq \left(\sum_{i=1}^n a_i^2\right)\left(\sum_{i=1}^n b_i^2\right)
	\] 
	with equality if and only if there exists a constant $k$ such that $a_i = k b_i$ for all $i$.
\end{theorem}
\begin{theorem}[2D Cauchy-Schwarz Inequality]
	\label{thm:2D_Cauchy-Schwarz_Inequality}
	For any real numbers $a, b, x, y$ we have
	\[
		(ax+by)^2 \leq (a^2+b^2)(x^2+y^2)
	\]
	with equality if and only if there exists a constant $k$ such that $a = k x$ and $b = k y$.
\end{theorem}
\begin{example}
	Try to find the maximum of the function
	\[
	2x+3y
	\] 
	subject to the constraint
	\[
	x^2+4y^2=4
	\] 
	\begin{answer}
		In order to use \ref{thm:2D_Cauchy-Schwarz_Inequality}, we hope
		\[
		Y = 2y, \quad bY = 3y
		\] 
		which gives $b = 3/2$, hence
		\[
			(2x+3y)^2 \leq (2^2+(3/2)^2)(x^2+(2y)^2) = 25
		\] 
		so the maximum of $2x+3y$ is $5$, and the equality holds when
		\[
			\frac{2}{2} = \frac{3/2}{2y} \implies y = 3/4
		\] 
	\end{answer}

	
\end{example}
\begin{example}
	Try to find the minimum of the function
	\[
		f(x,y,z) = \pi x^2 +y ^2 + \frac{\sqrt{3}}{4}z^2
	\] 
	subject to the constraint
	\[
	2\pi x + 4y + 3z  = 2
	\] 
	\begin{answer}
		Let
		\[
			X = \sqrt{\pi}x , \quad AX = 2\pi x 
		\] 
		\[
		Z = \sqrt{\sqrt{3}/4}z, \quad CZ = 3z
		\] 
		hence
		\[
			A = 2\sqrt{\pi}, \quad
			C = {3 / \sqrt{\sqrt{3}/4}} = 2 \cdot 3^{3 / 4}
		\] 
		by \ref{thm:Cauchy-Schwarz_Inequality}, we have
		\[
			4 = (2\pi x + 4y + 3z)^2 \leq (A^2+4^2+C^2)(X^2+y^2+Z^2) 
			= (4\pi + 16 + 12\sqrt{3})f(x,y,z)
		\] 
		so the minimum of $f(x,y,z)$ is
		\[
			f_{\min} = \frac{1}{\pi + 4 + 3\sqrt{3}}
		\] 
	\end{answer}
	
\end{example}
